1. Library Functions


These are built-in functions provided by C's standard library. They perform common tasks and are readily available for use.



  
example

  #include 
#include 

 int main() {
    double number = 49;
    double result = sqrt(number);  // sqrt() is a library function
    printf("Square root of %.2f is %.2f\n", number, result);
    return 0;
 }


output

 csharp
  Square root of 49.00 is 7.00


2. User-Defined Functions


These are functions created by the programmer to perform specific tasks.



  
example


  #include 

 int add(int a, int b) {
    return a + b;
 }

 int main() {
    int sum = add(5, 3);  // add() is a user-defined function
    printf("Sum is: %d\n", sum);
    return 0;
 }


ðŸ”„ Types of Functions Based on Arguments and Return Values


Functions can be categorized into four types:


1. Function with Arguments and Return Value



  #include 

 int multiply(int a, int b) {
    return a * b;
 }

  int main() {
    int result = multiply(4, 3);
    printf("Product is: %d\n", result);
    return 0;
   }
  
output

   Product is: 12

2. Function with Arguments and No Return Value



  #include 

   void printSum(int a, int b) {
    printf("Sum is: %d\n", a + b);
  }

  int main() {
    printSum(5, 3);
    return 0;
  }
  
output

  Sum is: 8


3. Function with No Arguments but with Return Value



  #include 

  int getNumber() {
    return 42;
  }

   int main() {
    int num = getNumber();
    printf("Number is: %d\n", num);
    return 0; 
   }
  
output

  Number is: 42


4. Function with No Arguments and No Return Value



  #include 

  void greet() {
    printf("Hello, World!\n");
  }

  int main() {
    greet();
    return 0;
  }

   
output

  Hello, World!
